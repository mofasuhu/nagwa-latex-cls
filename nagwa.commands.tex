% Typeset helper used at shipout: wraps content in a varwidth box and
% locally remaps \\ so bare \\ gets extra vertical space.

  \begingroup
    \begin{varwidth}[t]{\maxdimen}%
      \let\hw@origbs=\\%
      \def\\{\nagwa@hw@newline}%
      #1%
    \end{varwidth}%
  \endgroup
}

% 1. Core rendering logic
\def\@angle@render#1{%
\ifthenelse{\equal{\NagwaAngle}{en1}}{\oldangle{#1}}{}%
\ifthenelse{\equal{\NagwaAngle}{en2}}{\widehat{#1}}{}%
\ifthenelse{\equal{\NagwaAngle}{en3}}{\sphericalangle{#1}}{}%

}

% 2. Handler specifically for \left ... \right blocks
% It grabs \left, the left delimiter (#1), the content up to \right, and the right delimiter (#2)
\def\@angle@handle@left\left#1\right#2{%
\@angle@render{\left#1\right#2}%
}

% 3. Handler for commands with a braced argument (e.g., \text{ABC})
\def\@angle@unbraced@cmd#1#2{\@angle@render{#1{#2}}}

% 4. Handler for standard unbraced tokens (e.g., A, B, or \text)
\def\@angle@handle@normal#1{%
\@ifnextchar\bgroup{\@angle@unbraced@cmd{#1}}{\@angle@render{#1}}%
}

% 5. Logic to detect if the first unbraced token is \left
\def\@angle@unbraced#1{%
\ifx#1\left
    \expandafter\@firstoftwo
\else
    \expandafter\@secondoftwo
\fi
{\@angle@handle@left\left}%
{\@angle@handle@normal{#1}}%
}

% 6. Redefine \angle to peek ahead
\renewcommand{\angle}{%
\@ifnextchar\bgroup{\@angle@render}{\@angle@unbraced}%
}

% Command to convert each digit in page number to Hindi digits
\newcommand{\hindipagenumber}{%
    \directlua{
        local page_str = "\luaescapestring{\thepage}"
        local hindi_digits = {
            ["0"] = "٠", ["1"] = "١", ["2"] = "٢", ["3"] = "٣", ["4"] = "٤",
            ["5"] = "٥", ["6"] = "٦", ["7"] = "٧", ["8"] = "٨", ["9"] = "٩"
        }
        local result = ""
        for i = 1, string.len(page_str) do
            local c = page_str:sub(i,i)
            result = result .. (hindi_digits[c] or c)
        end
        tex.sprint(result)
    }%
}


% Command to convert total page count to Hindi digits
% NOTE: \pageref is protected (non-expandable) and cannot be used inside
% \directlua. Instead, we use Lua's token.get_macro to read the \r@LastPage
% label data directly and extract the page number with a Lua pattern.
\newcommand{\hinditotalpages}{%
    \directlua{
        local ref = ""
        pcall(function() ref = token.get_macro("r@LastPage") or "" end)
        local page = ref:match("^{.-}{(.-)}")
        if page and string.len(page) > 0 then
            local hindi_digits = {
                ["0"] = "٠", ["1"] = "١", ["2"] = "٢", ["3"] = "٣", ["4"] = "٤",
                ["5"] = "٥", ["6"] = "٦", ["7"] = "٧", ["8"] = "٨", ["9"] = "٩"
            }
            local result = ""
            for i = 1, string.len(page) do
                local c = page:sub(i,i)
                result = result .. (hindi_digits[c] or c)
            end
            tex.sprint(result)
        else
            tex.sprint("??")
        end
    }%
}
